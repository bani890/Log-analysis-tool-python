1. create view path_hits as select substr(log.path, 10) as slughit, count(path) as hits from log group by path order by hits desc; 

2. create view auth_slug_hit as select author,slughit, hits from articles,path_hits where articles.slug=path_hits.slughit;

3. create view error_table as select date(time), count(*) as error_hits from log where status='404 NOT FOUND' group by date(time) order by date(time);

4.create view all_views as select date(time), count(*) as views from log group by date(time) order by date(time);
